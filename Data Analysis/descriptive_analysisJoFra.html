#Loading the prepared data set

the_onexy <- readRDS('/Users/johannaschillig/Downloads/file_r.rds')
### Chi^2 test of independence

# Checking independence of response from grammatical condition
reponse_cond_xchart <- xtabs(~the_onexy$responded_gender + ~the_onexy$condition)                
print(reponse_cond_xchart)
##                           the_onexy$condition
## the_onexy$responded_gender gramm_female gramm_male gramm_neutral
##            female_response          664        293           484
##            male_response            155        492           324
chisq.test(
  reponse_cond_xchart, 
  correct = FALSE
)
## 
##  Pearson's Chi-squared test
## 
## data:  reponse_cond_xchart
## X-squared = 319, df = 2, p-value < 2.2e-16
# Checking independence of response from stereotype
reponse_stereo_xchart <- xtabs(~the_onexy$responded_gender + ~the_onexy$stereotype)                
print(reponse_stereo_xchart)
##                           the_onexy$stereotype
## the_onexy$responded_gender stereo_female stereo_male stereo_neutral
##            female_response           584         376            481
##            male_response             220         428            323
chisq.test(
  reponse_stereo_xchart, 
  correct = FALSE
)
## 
##  Pearson's Chi-squared test
## 
## data:  reponse_stereo_xchart
## X-squared = 111.87, df = 2, p-value < 2.2e-16
### Plotting the responses for different conditions ###

# grammatical neutral form

distribution_GNL <- the_onexy%>%
select(condition, responded_gender, stereotype)%>%
filter(condition == "gramm_neutral")
print(distribution_GNL)
## # A tibble: 808 × 3
##    condition     responded_gender stereotype    
##    <chr>         <chr>            <chr>         
##  1 gramm_neutral female_response  stereo_neutral
##  2 gramm_neutral female_response  stereo_male   
##  3 gramm_neutral male_response    stereo_female 
##  4 gramm_neutral female_response  stereo_male   
##  5 gramm_neutral female_response  stereo_female 
##  6 gramm_neutral male_response    stereo_neutral
##  7 gramm_neutral male_response    stereo_neutral
##  8 gramm_neutral female_response  stereo_male   
##  9 gramm_neutral female_response  stereo_male   
## 10 gramm_neutral male_response    stereo_female 
## # … with 798 more rows
distribution_GNL$stereotype <- factor(distribution_GNL$stereotype,     
                                     levels = c("stereo_male",
                                                "stereo_neutral",                                                                                "stereo_female"))

cond_gender_plot1 <- ggplot(data = distribution_GNL, mapping = aes(x = responded_gender, fill = responded_gender)) +
  geom_bar() +
  scale_fill_manual(values=c("deeppink4", "deepskyblue4"))+
  facet_wrap( ~ stereotype) +
    labs(
    x = 'Responded Gender',
    y = 'Count',
    title = "Distribution of responses for GNL depending on stereotypes") 
print(cond_gender_plot1)
